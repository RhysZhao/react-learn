function useRouter() {
  const { dispatch } = useContext(RouterStateContext);

  const navigate = useCallback(
    (url) => {
      dispatch({ type: 'navigate', url });

    [dispatch]
  );

  }, [dispatch]);

  return {
    navigate,
    goBack
  };
}
